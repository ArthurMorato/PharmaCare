```java
import jakarta.validation.constraints.*;

import java.time.LocalDate;

/**
 * Command para adicionar estoque de um produto em uma filial.
 * 
 * Pode criar um novo item de inventário ou atualizar um existente.
 */
public record AddStockCommand(
    
    @NotBlank(message = "ID do produto é obrigatório")
    @Pattern(regexp = "^PROD-[A-Z0-9]{10}$", message = "ID do produto deve seguir o padrão PROD-XXXXXXXXXX")
    String productId,
    
    @NotBlank(message = "ID da filial é obrigatório")
    @Pattern(regexp = "^BRANCH-[A-Z0-9]{8}$", message = "ID da filial deve seguir o padrão BRANCH-XXXXXXXX")
    String branchId,
    
    @NotNull(message = "Quantidade é obrigatória")
    @Min(value = 1, message = "Quantidade deve ser maior que zero")
    @Max(value = 10000, message = "Quantidade não pode exceder 10000 unidades")
    Integer quantity,
    
    @NotBlank(message = "Número do lote é obrigatório")
    @Size(min = 3, max = 50, message = "Número do lote deve ter entre 3 e 50 caracteres")
    String batchNumber,
    
    @NotNull(message = "Data de validade é obrigatória")
    @JsonFormat(pattern = "yyyy-MM-dd")
    @Future(message = "Data de validade deve ser futura")
    LocalDate expirationDate,
    
    @NotNull(message = "Custo unitário é obrigatório")
    @DecimalMin(value = "0.01", message = "Custo unitário deve ser maior que zero")
    @DecimalMax(value = "999999.99", message = "Custo unitário não pode exceder 999999.99")
    Double unitCost,
    
    @NotBlank(message = "Fornecedor é obrigatório")
    @Size(min = 2, max = 100, message = "Fornecedor deve ter entre 2 e 100 caracteres")
    String supplier,
    
    @NotBlank(message = "Nota fiscal é obrigatória")
    @Size(min = 1, max = 50, message = "Nota fiscal deve ter entre 1 e 50 caracteres")
    String invoiceNumber,
    
    @JsonFormat(pattern = "yyyy-MM-dd")
    LocalDate invoiceDate,
    
    @NotNull(message = "Tipo de movimentação é obrigatório")
    @Pattern(regexp = "^(PURCHASE|RETURN|ADJUSTMENT|TRANSFER_RECEIVED)$", 
             message = "Tipo de movimentação deve ser PURCHASE, RETURN, ADJUSTMENT ou TRANSFER_RECEIVED")
    String movementType,
    
    @Size(max = 500, message = "Observações não podem exceder 500 caracteres")
    String notes,
    
    @NotBlank(message = "Usuário responsável é obrigatório")
    @Size(min = 1, max = 100, message = "Usuário responsável deve ter entre 1 e 100 caracteres")
    String performedBy
    
) {
    
    /**
     * Valida regras customizadas.
     */
    public void validateCustomRules() {
        // Data da nota fiscal não pode ser futura
        if (invoiceDate != null && invoiceDate.isAfter(LocalDate.now())) {
            throw new IllegalArgumentException("Data da nota fiscal não pode ser futura");
        }
        
        // Data de validade deve ser pelo menos 30 dias no futuro
        if (expirationDate != null) {
            LocalDate minExpirationDate = LocalDate.now().plusDays(30);
            if (expirationDate.isBefore(minExpirationDate)) {
                throw new IllegalArgumentException(
                    "Data de validade deve ser pelo menos 30 dias no futuro. Data mínima: " + minExpirationDate
                );
            }
        }
        
        // Validar quantidade baseada no tipo de movimento
        validateMovementTypeRules();
    }
    
    private void validateMovementTypeRules() {
        // Para TRANSFER_RECEIVED, a quantidade deve ser positiva
        if ("TRANSFER_RECEIVED".equals(movementType) && quantity <= 0) {
            throw new IllegalArgumentException("Para transferências recebidas, a quantidade deve ser positiva");
        }
    }
    
    /**
     * Factory method para compras.
     */
    public static AddStockCommand forPurchase(
            String productId, String branchId, Integer quantity,
            String batchNumber, LocalDate expirationDate, Double unitCost,
            String supplier, String invoiceNumber, LocalDate invoiceDate,
            String performedBy) {
        
        return new AddStockCommand(
            productId, branchId, quantity, batchNumber, expirationDate,
            unitCost, supplier, invoiceNumber, invoiceDate,
            "PURCHASE", "Compra de fornecedor", performedBy
        );
    }
    
    /**
     * Factory method para devoluções.
     */
    public static AddStockCommand forReturn(
            String productId, String branchId, Integer quantity,
            String batchNumber, LocalDate expirationDate, String invoiceNumber,
            String performedBy) {
        
        return new AddStockCommand(
            productId, branchId, quantity, batchNumber, expirationDate,
            0.0, "Cliente", invoiceNumber, LocalDate.now(),
            "RETURN", "Devolução de cliente", performedBy
        );
    }
    
    /**
     * Factory method para ajustes positivos.
     */
    public static AddStockCommand forPositiveAdjustment(
            String productId, String branchId, Integer quantity,
            String batchNumber, LocalDate expirationDate, String notes,
            String performedBy) {
        
        return new AddStockCommand(
            productId, branchId, quantity, batchNumber, expirationDate,
            0.0, "Sistema", "AJUST-" + System.currentTimeMillis(), LocalDate.now(),
            "ADJUSTMENT", notes, performedBy
        );
    }
```

```java
    String productId,

    @Min(value = 1, message = "Quantidade deve ser maior que zero")
    @Max(value = 10000, message = "Quantidade não pode exceder 10000 unidades")
    Integer quantity,
    
    @NotNull(message = "Motivo da remoção é obrigatório")
    @Pattern(regexp = "^(SALE|TRANSFER_OUT|EXPIRED|DAMAGED|ADJUSTMENT)$",

    String reason,
    
    @NotBlank(message = "Número do documento é obrigatório")

    @Size(max = 100, message = "Número da prescrição não pode exceder 100 caracteres")
    String prescriptionNumber,
    
    @NotNull(message = "Valor unitário da venda é obrigatório")
    @DecimalMin(value = "0.01", message = "Valor unitário deve ser maior que zero")
    @DecimalMax(value = "999999.99", message = "Valor unitário não pode exceder 999999.99")
    Double unitPrice,
    
    @Size(max = 500, message = "Observações não podem exceder 500 caracteres")
    String notes,
    
    @NotBlank(message = "Usuário responsável é obrigatório")

    String performedBy
    
) {

     * Valida regras customizadas.
     */
    public void validateCustomRules() {

            throw new IllegalArgumentException("Para vendas, o preço unitário deve ser maior que zero");

        if ("TRANSFER_OUT".equals(reason) && unitPrice != 0) {
            throw new IllegalArgumentException("Para transferências, o preço unitário deve ser zero");
        }
        
        // Validar que não estamos tentando remover estoque negativo
        if (quantity <= 0) {
            throw new IllegalArgumentException("Quantidade para remoção deve ser maior que zero");
        }
    }
    
    /**
     * Factory method para vendas.
```

```java
package com.pharmacare.application.inventory.commands;

import jakarta.validation.constraints.*;
import com.fasterxml.jackson.annotation.JsonFormat;

import java.time.LocalDateTime;

/**
 * Command para transferir estoque entre filiais.
 * 
 * Implementa saga pattern para consistência eventual.
 */
public record TransferStockCommand(
    
    @NotBlank(message = "ID do produto é obrigatório")
    @Pattern(regexp = "^PROD-[A-Z0-9]{10}$", message = "ID do produto deve seguir o padrão PROD-XXXXXXXXXX")
    String productId,
    
    @NotBlank(message = "Filial de origem é obrigatória")
    @Pattern(regexp = "^BRANCH-[A-Z0-9]{8}$", message = "ID da filial deve seguir o padrão BRANCH-XXXXXXXX")
    String fromBranchId,
    
    @NotBlank(message = "Filial de destino é obrigatória")
    @Pattern(regexp = "^BRANCH-[A-Z0-9]{8}$", message = "ID da filial deve seguir o padrão BRANCH-XXXXXXXX")
    String toBranchId,
    
    @NotNull(message = "Quantidade é obrigatória")
    @Min(value = 1, message = "Quantidade deve ser maior que zero")
    @Max(value = 1000, message = "Quantidade não pode exceder 1000 unidades por transferência")
    Integer quantity,
    
    @NotBlank(message = "Número do lote é obrigatório")
    @Size(min = 3, max = 50, message = "Número do lote deve ter entre 3 e 50 caracteres")
    String batchNumber,
    
    @NotNull(message = "Data de validade é obrigatória")
    @JsonFormat(pattern = "yyyy-MM-dd")
    LocalDateTime expirationDate,
    
    @NotNull(message = "Custo unitário é obrigatório")
    @DecimalMin(value = "0.01", message = "Custo unitário deve ser maior que zero")
    @DecimalMax(value = "999999.99", message = "Custo unitário não pode exceder 999999.99")
    Double unitCost,
    
    @NotBlank(message = "Número da requisição é obrigatório")
    @Size(min = 1, max = 50, message = "Número da requisição deve ter entre 1 e 50 caracteres")
    String requestNumber,
    
    @Size(max = 500, message = "Observações não podem exceder 500 caracteres")
    String notes,
    
    @NotBlank(message = "Usuário solicitante é obrigatório")
    @Size(min = 1, max = 100, message = "Usuário solicitante deve ter entre 1 e 100 caracteres")
    String requestedBy,
    
    @NotNull(message = "Prazo de entrega é obrigatório")
    @JsonFormat(pattern = "yyyy-MM-dd")
    LocalDateTime expectedDeliveryDate,
    
    @NotNull(message = "Prioridade é obrigatória")
    @Pattern(regexp = "^(LOW|MEDIUM|HIGH|URGENT)$", 
             message = "Prioridade deve ser LOW, MEDIUM, HIGH ou URGENT")
    String priority,
    
    @NotNull(message = "Transportadora é obrigatória")
    @Size(min = 1, max = 100, message = "Transportadora deve ter entre 1 e 100 caracteres")
    String transporter
    
) {
    
    /**
     * Valida regras customizadas.
     */
    public void validateCustomRules() {
        // Filial de origem não pode ser igual à filial de destino
        if (fromBranchId.equals(toBranchId)) {
            throw new IllegalArgumentException("Filial de origem não pode ser igual à filial de destino");
        }
        
        // Data de entrega esperada deve ser futura
        if (expectedDeliveryDate.isBefore(LocalDateTime.now().plusHours(1))) {
            throw new IllegalArgumentException("Prazo de entrega deve ser pelo menos 1 hora no futuro");
        }
        
        // Data de validade deve ser futura e após a data de entrega
        if (expirationDate.isBefore(expectedDeliveryDate)) {
            throw new IllegalArgumentException(
                "Data de validade deve ser após a data de entrega esperada"
            );
        }
        
        // Para prioridade URGENT, quantidade máxima é 100
        if ("URGENT".equals(priority) && quantity > 100) {
            throw new IllegalArgumentException("Para transferências urgentes, quantidade máxima é 100 unidades");
        }
    }
    
    /**
     * Factory method para transferência urgente.
     */
    public static TransferStockCommand urgentTransfer(
            String productId, String fromBranchId, String toBranchId,
            Integer quantity, String batchNumber, LocalDateTime expirationDate,
            Double unitCost, String requestNumber, String requestedBy,
            String transporter) {
        
        return new TransferStockCommand(
            productId, fromBranchId, toBranchId, quantity,
            batchNumber, expirationDate, unitCost, requestNumber,
            "Transferência urgente", requestedBy,
            LocalDateTime.now().plusHours(4), "URGENT", transporter
        );
    }
    
    /**
     * Factory method para transferência normal.
     */
    public static TransferStockCommand normalTransfer(
            String productId, String fromBranchId, String toBranchId,
            Integer quantity, String batchNumber, LocalDateTime expirationDate,
            Double unitCost, String requestNumber, String requestedBy) {
        
        return new TransferStockCommand(
            productId, fromBranchId, toBranchId, quantity,
            batchNumber, expirationDate, unitCost, requestNumber,
            "Transferência normal", requestedBy,
            LocalDateTime.now().plusDays(2), "MEDIUM", "Transporte interno"
        );
    }
    
    /**
     * Gera ID único para a transferência.
     */
    public String generateTransferId() {
        return String.format("TRANSFER-%s-%s-%s-%d",
            productId.substring(0, 8),
            fromBranchId.substring(0, 8),
            toBranchId.substring(0, 8),
            System.currentTimeMillis() % 10000
        );
    }
}
```

```java
package com.pharmacare.application.inventory.responses;

import com.fasterxml.jackson.annotation.JsonFormat;

import java.math.BigDecimal;
import java.time.LocalDateTime;
import java.util.List;

/**
 * Response para alertas de estoque baixo.
 */
public record LowStockAlertResponse(
    
    String alertId,
    
    String productId,
    
    String productName,
    
    String branchId,
    
    String branchName,
    
    Integer currentStock,
    
    Integer minimumStock,
    
    Integer reorderPoint,
    
    Integer suggestedReorderQuantity,
    
    BigDecimal lastPurchasePrice,
    
    Integer daysSinceLastPurchase,
    
    Double averageDailySales,
    
    Integer estimatedDaysUntilStockout,
    
    String severity,
    
    List<String> recommendedActions,
    
    @JsonFormat(pattern = "yyyy-MM-dd'T'HH:mm:ss")
    LocalDateTime firstAlertDate,
    
    @JsonFormat(pattern = "yyyy-MM-dd'T'HH:mm:ss")
    LocalDateTime lastAlertDate,
    
    Integer alertCount,
    
    Boolean acknowledged,
    
    String acknowledgedBy,
    
    @JsonFormat(pattern = "yyyy-MM-dd'T'HH:mm:ss")
    LocalDateTime acknowledgedAt,
    
    String notes
    
) {
    
    /**
     * Calcula a severidade baseada no nível de estoque.
     */
    public static String calculateSeverity(Integer currentStock, Integer minimumStock) {
        if (currentStock == null || minimumStock == null) {
            return "UNKNOWN";
        }
        
        double percentage = (double) currentStock / minimumStock;
        
        if (percentage <= 0.2) return "CRITICAL";
        if (percentage <= 0.5) return "HIGH";
        if (percentage <= 0.8) return "MEDIUM";
        return "LOW";
    }
    
    /**
     * Calcula quantidade sugerida para reposição.
     */
    public static Integer calculateSuggestedReorderQuantity(
            Integer minimumStock, Integer currentStock, 
            Double averageDailySales, Integer leadTimeDays) {
        
        int safetyStock = (int) Math.ceil(averageDailySales * leadTimeDays * 1.5);
        int reorderQuantity = minimumStock + safetyStock - currentStock;
        
        return Math.max(reorderQuantity, minimumStock);
    }
    
    /**
     * Factory method para criar alerta.
     */
    public static LowStockAlertResponse createAlert(
            String productId, String productName, String branchId, String branchName,
            Integer currentStock, Integer minimumStock, Double averageDailySales) {
        
        String severity = calculateSeverity(currentStock, minimumStock);
        int estimatedDays = averageDailySales > 0 ? 
            (int) Math.floor(currentStock / averageDailySales) : 999;
        
        List<String> actions = List.of(
            "Verificar fornecedores disponíveis",
            "Avaliar transferência de outras filiais",
            "Priorizar pedido de compra"
        );
        
        return new LowStockAlertResponse(
            "ALERT-" + System.currentTimeMillis(),
            productId,
            productName,
            branchId,
            branchName,
            currentStock,
```

```java
        log.info("Adicionando estoque: produto={}, filial={}, quantidade={}", 
                command.productId(), command.branchId(), command.quantity());
        
        try {
            // 1. Validação
            validateCommand(command);
            validateProductAndBranch(command.productId(), command.branchId());
            
            // 2. Buscar ou criar item de inventário
            InventoryItem inventoryItem = findOrCreateInventoryItem(
                command.productId(), command.branchId(), command.batchNumber()
            );

            Quantity quantity = Quantity.of(command.quantity());
            BatchNumber batchNumber = BatchNumber.from(command.batchNumber());
            ExpirationDate expirationDate = ExpirationDate.from(command.expirationDate());
            
            inventoryItem.addStock(
                quantity,
                batchNumber,
                expirationDate,
                command.unitCost(),
                command.supplier(),
                command.invoiceNumber(),
                command.invoiceDate(),
                command.movementType(),
                command.notes(),
                command.performedBy()
            );

            publishInventoryEvents(savedItem, "STOCK_ADDED");
            
            // 6. Retornar response
            return buildStockLevelResponse(savedItem);

        try {
            // 1. Validação

                    BusinessException.BusinessErrorCode.RESOURCE_NOT_FOUND,

            // 3. Validar estoque suficiente
            validateSufficientStock(inventoryItem, command.quantity(), command.batchNumber());
            
            // 4. Remover estoque
            inventoryItem.removeStock(
                Quantity.of(command.quantity()),
                command.reason(),
                command.documentNumber(),
                command.customerId(),
                command.prescriptionNumber(),
                command.unitPrice(),
                command.notes(),
                command.performedBy()
            );

            // 7. Verificar se precisa de alerta

            throw e;

            List<String> productIdStrings = productIds.stream()

                .findAllByProductIdsAndBranchId(productIdStrings, branchId.value());

                .map(this::buildStockLevelResponse)
                .collect(Collectors.toList());

            Set<String> foundProductIds = inventoryItems.stream()
                .map(item -> item.getProductId().value())
                .collect(Collectors.toSet());
            
            productIdStrings.stream()
                .filter(productId -> !foundProductIds.contains(productId))
                .forEach(productId -> responses.add(
                    buildEmptyStockLevelResponse(productId, branchId.value())

            return responses;

    private void validateProductAndBranch(String productId, String branchId) {
        // Validar produto existe
        Product product = productRepository.findById(productId)
            .orElseThrow(() -> new BusinessException(
                BusinessException.BusinessErrorCode.RESOURCE_NOT_FOUND,
                "Produto não encontrado: " + productId
            ));
        
        // Validar produto está ativo
        if (!product.isActive()) {
            throw new BusinessException(
                BusinessException.BusinessErrorCode.OPERATION_NOT_ALLOWED,
                "Produto não está ativo: " + productId

        boolean branchExists = branchRepository.existsByIdAndActive(branchId, true);
        if (!branchExists) {
            throw new BusinessException(
                BusinessException.BusinessErrorCode.RESOURCE_NOT_FOUND,
                "Filial não encontrada ou inativa: " + branchId
```

```java
package com.pharmacare.application.inventory.handlers.impl.saga;

import com.pharmacare.application.inventory.commands.TransferStockCommand;
import com.pharmacare.application.inventory.responses.TransferStockResponse;
import com.pharmacare.application.inventory.handlers.impl.saga.events.*;
import com.pharmacare.domain.core.inventory.aggregates.InventoryItem;
import com.pharmacare.domain.core.inventory.ports.InventoryItemRepository;
import lombok.extern.slf4j.Slf4j;
import org.springframework.context.ApplicationEventPublisher;
import org.springframework.retry.annotation.Backoff;
import org.springframework.retry.annotation.Retryable;
import org.springframework.scheduling.annotation.Async;
import org.springframework.stereotype.Service;
import org.springframework.transaction.annotation.Transactional;

import java.time.LocalDateTime;
import java.util.Map;
import java.util.concurrent.ConcurrentHashMap;

/**
 * Saga para gerenciar transferências de estoque entre filiais.
 * 
 * Implementa compensação para rollback em caso de falha.
 */
@Slf4j
@Service
public class InventoryTransferSaga {
    
    private final InventoryItemRepository inventoryItemRepository;
    private final ApplicationEventPublisher eventPublisher;
    
    // Estado das sagas em memória (em produção, usar Redis ou banco)
    private final Map<String, TransferSagaState> sagaStates = new ConcurrentHashMap<>();
    
    // Passos da saga
    private static final String[] SAGA_STEPS = {
        "VALIDATION",
        "RESERVATION",
        "PICKING",
        "PACKING",
        "DISPATCH",
        "TRANSPORT",
        "RECEIVING",
        "CONFIRMATION"
    };
    
    public InventoryTransferSaga(
            InventoryItemRepository inventoryItemRepository,
            ApplicationEventPublisher eventPublisher) {
        
        this.inventoryItemRepository = inventoryItemRepository;
        this.eventPublisher = eventPublisher;
    }
    
    /**
     * Inicia uma nova saga de transferência.
     */
    @Transactional
    public TransferStockResponse initiateTransfer(String transferId, TransferStockCommand command) {
        log.info("Iniciando saga de transferência: {}", transferId);
        
        // 1. Criar estado da saga
        TransferSagaState sagaState = new TransferSagaState(transferId, command);
        sagaStates.put(transferId, sagaState);
        
        // 2. Publicar evento de início
        eventPublisher.publishEvent(new TransferInitiatedEvent(
            transferId, command, LocalDateTime.now()
        ));
        
        // 3. Executar saga de forma assíncrona
        executeSagaAsync(sagaState);
        
        // 4. Retornar response inicial
        return TransferStockResponse.initiated(
            transferId, command,
            "Produto " + command.productId(),
            "Filial " + command.fromBranchId(),
            "Filial " + command.toBranchId()
        );
    }
    
    /**
     * Executa a saga de forma assíncrona.
     */
    @Async
    @Retryable(
        value = {RuntimeException.class},
        maxAttempts = 3,
        backoff = @Backoff(delay = 1000, multiplier = 2)
    )
    public void executeSagaAsync(TransferSagaState sagaState) {
        try {
            executeSaga(sagaState);
        } catch (Exception e) {
            log.error("Erro na execução da saga {}: {}", 
                    sagaState.getTransferId(), e.getMessage(), e);
            
            // Tentar compensação
            compensateSaga(sagaState, e.getMessage());
        }
    }
    
    /**
     * Executa todos os passos da saga.
     */
    private void executeSaga(TransferSagaState sagaState) {
        String transferId = sagaState.getTransferId();
        TransferStockCommand command = sagaState.getCommand();
        
        log.info("Executando saga {}: de {} para {}, produto={}, quantidade={}", 
                transferId, command.fromBranchId(), command.toBranchId(),
                command.productId(), command.quantity());
        
        try {
            // Executar cada passo sequencialmente
            for (String stepName : SAGA_STEPS) {
                if (sagaState.isCompleted()) {
                    break;
                }
                
                sagaState.setCurrentStep(stepName);
                TransferSagaState.SagaStep step = new TransferSagaState.SagaStep(stepName);
                sagaState.addStep(step);
                
                step.start();
                log.info("Saga {}: Iniciando passo {}", transferId, stepName);
                
                try {
                    executeStep(stepName, sagaState, step);
                    step.complete("Sucesso");
                    log.info("Saga {}: Passo {} completado", transferId, stepName);
                    
                } catch (Exception e) {
                    step.fail(e.getMessage());
                    log.error("Saga {}: Falha no passo {}: {}", 
                            transferId, stepName, e.getMessage(), e);
                    
                    throw e; // Propagar para compensação
                }
            }
            
            // Se chegou aqui, todos os passos foram completados
            if (!sagaState.isCompleted()) {
                sagaState.complete();
                log.info("Saga {}: Transferência completada com sucesso", transferId);
                
                // Publicar evento de conclusão
                eventPublisher.publishEvent(new TransferCompletedEvent(
                    transferId, command, LocalDateTime.now()
                ));
            }
            
        } catch (Exception e) {
            sagaState.fail(e.getMessage());
            log.error("Saga {}: Falha na transferência: {}", transferId, e.getMessage());
            
            // Publicar evento de falha
            eventPublisher.publishEvent(new TransferFailedEvent(
                transferId, command, e.getMessage(), LocalDateTime.now()
            ));
            
            throw e;
        }
    }
    
    /**
     * Executa um passo específico da saga.
     */
    private void executeStep(
            String stepName, TransferSagaState sagaState, 
            TransferSagaState.SagaStep step) {
        
        String transferId = sagaState.getTransferId();
        TransferStockCommand command = sagaState.getCommand();
        
        switch (stepName) {
            case "VALIDATION":
                validateTransfer(command);
                break;
                
            case "RESERVATION":
                reserveStock(command);
                break;
                
            case "PICKING":
                // Simular picking
                log.debug("Saga {}: Picking do produto", transferId);
                Thread.sleep(500); // Simular delay
                break;
                
            case "PACKING":
                // Simular packing
                log.debug("Saga {}: Packing do produto", transferId);
                Thread.sleep(500);
                break;
                
            case "DISPATCH":
                // Simular dispatch
                log.debug("Saga {}: Dispatch para transporte", transferId);
                Thread.sleep(1000);
                break;
                
            case "TRANSPORT":
                // Simular transporte
                log.debug("Saga {}: Em transporte", transferId);
                Thread.sleep(2000);
                break;
                
            case "RECEIVING":
                receiveStock(command);
                break;
                
            case "CONFIRMATION":
                // Simular confirmação
                log.debug("Saga {}: Aguardando confirmação", transferId);
                Thread.sleep(1000);
                break;
                
            default:
                throw new IllegalArgumentException("Passo desconhecido: " + stepName);
        }
    }
    
    /**
     * Valida a transferência.
     */
    private void validateTransfer(TransferStockCommand command) {
        // Validar estoque suficiente na origem
        InventoryItem sourceItem = inventoryItemRepository
            .findByProductIdAndBranchId(command.productId(), command.fromBranchId())
            .orElseThrow(() -> new IllegalStateException(
                "Produto não encontrado na filial de origem"
            ));
        
        if (sourceItem.getAvailableQuantity() < command.quantity()) {
            throw new IllegalStateException(
                String.format("Estoque insuficiente na origem. Disponível: %d, Solicitado: %d",
                    sourceItem.getAvailableQuantity(), command.quantity())
            );
        }
        
        // Validar data de validade
        if (command.expirationDate().isBefore(LocalDateTime.now().plusDays(7))) {
            throw new IllegalStateException(
                "Produto com validade muito próxima para transferência"
            );
        }
        
        log.debug("Validação da transferência aprovada");
    }
    
    /**
     * Reserva estoque na origem.
     */
    @Transactional
    private void reserveStock(TransferStockCommand command) {
        InventoryItem sourceItem = inventoryItemRepository
            .findByProductIdAndBranchId(command.productId(), command.fromBranchId())
            .orElseThrow(() -> new IllegalStateException(
                "Produto não encontrado na filial de origem"
            ));
        
        // Reservar estoque
        sourceItem.reserveStock(command.quantity(), "TRANSFER-" + command.requestNumber());
        inventoryItemRepository.save(sourceItem);

    /**
     * Recebe estoque no destino.
     */
    @Transactional
    private void receiveStock(TransferStockCommand command) {
        // Criar ou atualizar item no destino
        InventoryItem destinationItem = inventoryItemRepository
            .findByProductIdAndBranchId(command.productId(), command.toBranchId())
            .orElseGet(() -> InventoryItem.create(
                com.pharmacare.domain.core.inventory.valueobjects.ProductId.from(command.productId()),
                com.pharmacare.domain.core.inventory.valueobjects.BranchId.from(command.toBranchId()),
                command.batchNumber()
            ));
        
        // Adicionar estoque recebido
        destinationItem.addStock(

            com.pharmacare.domain.core.inventory.valueobjects.BatchNumber.from(command.batchNumber()),

            "TRANSFER_RECEIVED",
            "Transferência da filial " + command.fromBranchId(),
            "Sistema"
        );
        
        inventoryItemRepository.save(destinationItem);
        
        // Liberar reserva na origem
        InventoryItem sourceItem = inventoryItemRepository
            .findByProductIdAndBranchId(command.productId(), command.fromBranchId())
            .orElseThrow(() -> new IllegalStateException(
                "Produto não encontrado na filial de origem para liberar reserva"
            ));
        
        sourceItem.releaseReservation("TRANSFER-" + command.requestNumber(), command.quantity());
        inventoryItemRepository.save(sourceItem);
        
        log.debug("Estoque recebido no destino: {} unidades", command.quantity());
    }
    
    /**
     * Compensa a saga em caso de falha.
     */
    private void compensateSaga(TransferSagaState sagaState, String failureReason) {
        log.info("Iniciando compensação da saga {}: {}", 
                sagaState.getTransferId(), failureReason);
        
        try {
            // Executar passos de compensação na ordem inversa
            for (int i = sagaState.getSteps().size() - 1; i >= 0; i--) {
                TransferSagaState.SagaStep step = sagaState.getSteps().get(i);
                
                if ("COMPLETED".equals(step.getStatus())) {
                    compensateStep(step.getName(), sagaState, step);
                    step.compensate("Compensado");
                }
            }
            
            log.info("Saga {}: Compensação completada", sagaState.getTransferId());
            
        } catch (Exception e) {
            log.error("Saga {}: Erro durante compensação: {}", 
                    sagaState.getTransferId(), e.getMessage(), e);
            
            // Em caso de erro na compensação, registrar manualmente
            sagaState.setFailureReason(failureReason + " | Erro na compensação: " + e.getMessage());
        }
    }
    
    /**
     * Compensa um passo específico.
     */
    private void compensateStep(
            String stepName, TransferSagaState sagaState, 
            TransferSagaState.SagaStep step) {
        
        String transferId = sagaState.getTransferId();
        TransferStockCommand command = sagaState.getCommand();
        
        log.debug("Saga {}: Compensando passo {}", transferId, stepName);
        
        switch (stepName) {
            case "RESERVATION":
                // Liberar reserva se foi feita
                try {
                    InventoryItem sourceItem = inventoryItemRepository
                        .findByProductIdAndBranchId(command.productId(), command.fromBranchId());
                    
                    if (sourceItem != null) {
                        sourceItem.releaseReservation(
                            "TRANSFER-" + command.requestNumber(), command.quantity());
                        inventoryItemRepository.save(sourceItem);
                        log.debug("Reserva liberada durante compensação");
                    }
                } catch (Exception e) {
                    log.warn("Erro ao liberar reserva durante compensação: {}", e.getMessage());
                }
                break;
                
            case "RECEIVING":
                // Remover estoque adicionado no destino
                try {
                    InventoryItem destinationItem = inventoryItemRepository
                        .findByProductIdAndBranchId(command.productId(), command.toBranchId());
                    
                    if (destinationItem != null && 
                        destinationItem.getTotalQuantity() >= command.quantity()) {
                        
                        destinationItem.removeStock(
                            com.pharmacare.domain.core.inventory.valueobjects.Quantity.of(
                                command.quantity()),
                            "COMPENSATION",
                            "COMP-" + command.requestNumber(),
                            "SISTEMA",
                            "",
                            0.0,
                            "Compensação de transferência falha",
                            "Sistema"
                        );
                        
                        inventoryItemRepository.save(destinationItem);
                        log.debug("Estoque removido do destino durante compensação");
                    }
                } catch (Exception e) {
                    log.warn("Erro ao remover estoque durante compensação: {}", e.getMessage());
                }
                break;
                
            default:
                // Outros passos não precisam de compensação específica
                break;
        }
    }
    
    /**
     * Obtém status da transferência.
     */
    public TransferStockResponse getTransferStatus(String transferId) {
        TransferSagaState sagaState = sagaStates.get(transferId);
        
        if (sagaState == null) {
            throw new IllegalStateException("Transferência não encontrada: " + transferId);
        }
        
        // Calcular progresso
        int completedSteps = (int) sagaState.getSteps().stream()
            .filter(step -> "COMPLETED".equals(step.getStatus()))
            .count();
        
        int totalSteps = SAGA_STEPS.length;
        int progress = (int) ((completedSteps / (double) totalSteps) * 100);
        
        // Construir response
        return TransferStockResponse.initiated(
            transferId,
            sagaState.getCommand(),
            "Produto " + sagaState.getCommand().productId(),
            "Filial " + sagaState.getCommand().fromBranchId(),
            "Filial " + sagaState.getCommand().toBranchId()
        ).withProgress(sagaState.getCurrentStep(), progress);
    }
    
    /**
     * Confirma recebimento da transferência.
     */
    @Transactional
    public TransferStockResponse confirmReceipt(
            String transferId, String confirmationCode, String receivedBy) {
        
        TransferSagaState sagaState = sagaStates.get(transferId);
        
        if (sagaState == null) {
            throw new IllegalStateException("Transferência não encontrada: " + transferId);
        }
        
        if (!"COMPLETED".equals(sagaState.getStatus())) {
            throw new IllegalStateException("Transferência não está completada");
        }
        
        // Atualizar estado
        sagaState.setStatus("RECEIVED_CONFIRMED");
        
        // Registrar confirmação
        TransferSagaState.SagaStep confirmationStep = new TransferSagaState.SagaStep("CONFIRMATION");
        confirmationStep.start();
        confirmationStep.complete("Confirmado por " + receivedBy);
        sagaState.addStep(confirmationStep);
        
        log.info("Transferência {} confirmada por {}", transferId, receivedBy);
        
        return getTransferStatus(transferId);
    }
    
    /**
     * Cancela uma transferência.
     */
    @Transactional
    public TransferStockResponse cancelTransfer(
            String transferId, String reason, String cancelledBy) {
        
        TransferSagaState sagaState = sagaStates.get(transferId);
        
        if (sagaState == null) {
            throw new IllegalStateException("Transferência não encontrada: " + transferId);
        }
        
        if (sagaState.isCompleted()) {
            throw new IllegalStateException("Transferência já foi completada ou falhou");
        }
        
        // Marcar como cancelada
        sagaState.fail("CANCELADO: " + reason);
        
        // Executar compensação
        compensateSaga(sagaState, "Cancelada por " + cancelledBy + ": " + reason);
        
        log.info("Transferência {} cancelada por {}: {}", 
                transferId, cancelledBy, reason);
        
        return getTransferStatus(transferId);
    }
}
```

```java
        return RedisCacheManager.builder(redisConnectionFactory)
            .cacheDefaults(defaultConfig)
            .withInitialCacheConfigurations(cacheConfigurations)
```

## **12. CONSIDERAÇÕES DE BACKWARD COMPATIBILITY**

### **12.1 Versionamento de Commands**
- Commands são imutáveis (records)
- Novos campos opcionais com valores default
- Validação forward-compatible

### **12.2 Cache Management**
- Versionamento de keys de cache
- TTL apropriado para diferentes dados
- Invalidação gradual durante migrações

### **12.3 Saga Pattern**
- Estados da saga serializáveis
- Compensação idempotente
- Logging detalhado para debug

### **12.4 Performance**
- Cache L1/L2 para operações frequentes
- Batch operations para grandes volumes
- Indexação apropriada no MongoDB

### **12.5 Monitoring**
- Métricas para operações de estoque
- Alertas para níveis críticos
- Tracing para sagas de transferência
