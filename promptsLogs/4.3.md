```java
import com.fasterxml.jackson.annotation.JsonTypeInfo;
import lombok.AllArgsConstructor;
import lombok.Builder;
import lombok.Data;
import lombok.NoArgsConstructor;

import java.time.LocalDateTime;
import java.util.HashMap;
import java.util.Map;

/**
 * Classe base para eventos de domínio.
 * Todos os eventos devem estender esta classe.
 */
@Data
@Builder
@NoArgsConstructor
@AllArgsConstructor
@JsonTypeInfo(use = JsonTypeInfo.Id.CLASS, property = "@type")
public abstract class DomainEvent {
    
    @Builder.Default
    private String eventId = UUID.randomUUID().toString();
    
    private String aggregateId;
    private String eventType;
    private String aggregateType;
    
    @Builder.Default
    private LocalDateTime timestamp = LocalDateTime.now();
    
    private Long version;
    private String correlationId;
    private String causationId;
    
    @Builder.Default
    private Map<String, Object> payload = new HashMap<>();
    
    @Builder.Default
    private Map<String, Object> metadata = new HashMap<>();
    
    /**
     * Adiciona metadado ao evento.
     */
    public void addMetadata(String key, Object value) {
        if (metadata == null) {
            metadata = new HashMap<>();
        }
        metadata.put(key, value);
    }
    
    /**
     * Adiciona dado ao payload.
     */
    public void addPayload(String key, Object value) {
        if (payload == null) {
            payload = new HashMap<>();
        }
        payload.put(key, value);
    }
}
```

```java
// infrastructure-adapter-output-events/src/main/java/com/pharmacare/infrastructure/adapter/output/events/adapter/EventPublisherKafkaAdapter.java
package com.pharmacare.infrastructure.adapter.output.events.adapter;

import com.fasterxml.jackson.core.JsonProcessingException;
import com.fasterxml.jackson.databind.ObjectMapper;
import com.pharmacare.domain.core.event.DomainEvent;
import com.pharmacare.domain.core.port.output.EventPublisherPort;
import com.pharmacare.infrastructure.adapter.output.events.dto.KafkaDeadLetterMessage;
import com.pharmacare.infrastructure.adapter.output.events.metrics.KafkaEventMetrics;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.slf4j.MDC;
import org.springframework.kafka.KafkaException;
import org.springframework.kafka.core.KafkaTemplate;
import org.springframework.kafka.support.KafkaHeaders;
import org.springframework.kafka.support.SendResult;
import org.springframework.retry.annotation.Backoff;
import org.springframework.retry.annotation.Retryable;
import org.springframework.scheduling.annotation.Async;
import org.springframework.stereotype.Component;
import org.springframework.util.concurrent.ListenableFuture;
import org.springframework.util.concurrent.ListenableFutureCallback;

import java.nio.charset.StandardCharsets;
import java.time.LocalDateTime;
import java.util.HashMap;
import java.util.Map;
import java.util.concurrent.CompletableFuture;

/**
 * Adapter Kafka para publicação de eventos de domínio.
 * Implementa o EventPublisherPort seguindo a arquitetura hexagonal.
 */
@Component
@RequiredArgsConstructor
@Slf4j
public class EventPublisherKafkaAdapter implements EventPublisherPort {
    
    private static final String PRODUCT_EVENTS_TOPIC = "product.events";
    private static final String INVENTORY_EVENTS_TOPIC = "inventory.events";
    private static final String DLQ_SUFFIX = ".dlq";
    
    private final KafkaTemplate<String, DomainEvent> kafkaTemplate;
    private final ObjectMapper objectMapper;
    private final KafkaEventMetrics kafkaEventMetrics;
    
    @Override
    public void publish(DomainEvent event) {
        publish(event, null);
    }
    
    @Override
    @Async("kafkaEventPublisherExecutor")
    @Retryable(
        value = {KafkaException.class, JsonProcessingException.class},
        maxAttempts = 3,
        backoff = @Backoff(delay = 1000, multiplier = 2)
    )
    public void publish(DomainEvent event, Map<String, String> headers) {
        String topic = resolveTopic(event);
        String key = event.getAggregateId();
        
        // Adiciona correlationId do MDC se disponível
        String correlationId = MDC.get("correlationId");
        if (correlationId != null && !correlationId.isEmpty()) {
            event.setCorrelationId(correlationId);
        }
        
        // Adiciona metadados de tracing
        event.addMetadata("publishedAt", LocalDateTime.now().toString());
        event.addMetadata("publisher", "kafka-adapter");
        event.addMetadata("application", "pharmacare-erp");
        
        log.debug("Publicando evento no tópico {}: eventId={}, type={}", 
            topic, event.getEventId(), event.getEventType());
        
        try {
            // Cria headers da mensagem
            org.springframework.messaging.Message<DomainEvent> message = 
                createMessage(event, topic, key, headers);
            
            // Envia de forma assíncrona com callback
            ListenableFuture<SendResult<String, DomainEvent>> future = 
                kafkaTemplate.send(message);
            
            future.addCallback(new ListenableFutureCallback<>() {
                @Override
                public void onSuccess(SendResult<String, DomainEvent> result) {
                    handleSuccess(event, topic, result);
                }
                
                @Override
                public void onFailure(Throwable ex) {
                    handleFailure(event, topic, key, ex);
                }
            });
            
        } catch (Exception ex) {
            log.error("Erro ao tentar publicar evento no tópico {}: {}", topic, event, ex);
            kafkaEventMetrics.incrementFailure(event.getEventType(), topic);
            throw new KafkaException("Falha ao publicar evento no Kafka", ex);
        }
    }
    
    /**
     * Determina o tópico correto com base no tipo de evento.
     */
    private String resolveTopic(DomainEvent event) {
        String eventType = event.getEventType();
        if (eventType.startsWith("product.")) {
            return PRODUCT_EVENTS_TOPIC;
        } else if (eventType.startsWith("inventory.")) {
            return INVENTORY_EVENTS_TOPIC;
        } else {
            // Tópico padrão para outros eventos
            return "domain.events";
        }
    }
    
    /**
     * Cria mensagem Kafka com headers.
     */
    private org.springframework.messaging.Message<DomainEvent> createMessage(
        DomainEvent event, 
        String topic, 
        String key,
        Map<String, String> customHeaders
    ) {
        return org.springframework.messaging.support.MessageBuilder
            .withPayload(event)
            .setHeader(KafkaHeaders.TOPIC, topic)
            .setHeader(KafkaHeaders.KEY, key)
            .setHeader(KafkaHeaders.TIMESTAMP, System.currentTimeMillis())
            .setHeader("event-id", event.getEventId())
            .setHeader("event-type", event.getEventType())
            .setHeader("correlation-id", event.getCorrelationId())
            .setHeader("aggregate-id", event.getAggregateId())
            .setHeader("aggregate-type", event.getAggregateType())
            .setHeader("version", event.getVersion())
            .setHeader("source", "pharmacare-erp")
            .setHeader("content-type", "application/json")
            .setHeader("encoding", StandardCharsets.UTF_8.name())
            .build();
    }
    
    /**
     * Manipula sucesso na publicação.
     */
    private void handleSuccess(
        DomainEvent event, 
        String topic, 
        SendResult<String, DomainEvent> result
    ) {
        kafkaEventMetrics.incrementSuccess(event.getEventType(), topic);
        
        log.debug("Evento publicado com sucesso no tópico {}: offset={}, partition={}", 
            topic,
            result.getRecordMetadata().offset(),
            result.getRecordMetadata().partition());
        
        // Log de métricas detalhado
        log.info("Evento {} publicado: eventId={}, topic={}, key={}", 
            event.getEventType(),
            event.getEventId(),
            topic,
            event.getAggregateId());
    }
    
    /**
     * Manipula falha na publicação.
     */
    private void handleFailure(
        DomainEvent event, 
        String topic, 
        String key,
        Throwable cause
    ) {
        log.error("Falha ao publicar evento no tópico {}: eventId={}, type={}", 
            topic, event.getEventId(), event.getEventType(), cause);
        
        kafkaEventMetrics.incrementFailure(event.getEventType(), topic);
        
        // Envia para Dead Letter Queue após falhas persistentes
        sendToDeadLetterQueue(topic, key, event, cause);
    }
    
    /**
     * Envia evento para Dead Letter Queue.
     */
    private void sendToDeadLetterQueue(
        String originalTopic, 
        String key, 
        DomainEvent event, 
        Throwable cause
    ) {
        try {
            String dlqTopic = originalTopic + DLQ_SUFFIX;
            
            KafkaDeadLetterMessage dlqMessage = KafkaDeadLetterMessage.builder()
                .originalTopic(originalTopic)
                .originalKey(key)
                .event(event)
                .failureCause(cause.getMessage())
                .failureTimestamp(LocalDateTime.now())
                .retryCount(3) // Máximo de retentativas
                .application("pharmacare-erp")
                .build();
            
            // Publica na DLQ com formato diferente (string JSON)
            kafkaTemplate.send(dlqTopic, key, objectMapper.writeValueAsString(dlqMessage));
            
            log.warn("Evento enviado para DLQ {}: eventId={}, cause={}", 
                dlqTopic, event.getEventId(), cause.getMessage());
            
            kafkaEventMetrics.incrementDlq(event.getEventType(), dlqTopic);
            
        } catch (Exception ex) {
            log.error("FALHA CRÍTICA: Não foi possível enviar evento para DLQ: eventId={}", 
                event.getEventId(), ex);
            
            // Em casos extremos, log em arquivo ou banco de dados
            logToErrorStorage(event, cause);
        }
    }
    
    /**
     * Log alternativo em caso de falha total.
     */
    private void logToErrorStorage(DomainEvent event, Throwable cause) {
        // Implementação para log em banco de dados ou arquivo
        log.error("EVENTO PERDIDO - eventId={}, type={}, aggregateId={}, cause={}",
            event.getEventId(),
            event.getEventType(),
            event.getAggregateId(),
            cause.getMessage());
        
        // TODO: Implementar persistência em banco de dados para eventos perdidos
    }
    
    /**
     * Método síncrono para casos que requerem confirmação imediata.
     */
    public CompletableFuture<SendResult<String, DomainEvent>> publishWithConfirmation(
        DomainEvent event
    ) {
        CompletableFuture<SendResult<String, DomainEvent>> future = new CompletableFuture<>();
        
        String topic = resolveTopic(event);
        String key = event.getAggregateId();
        
        ListenableFuture<SendResult<String, DomainEvent>> kafkaFuture = 
            kafkaTemplate.send(topic, key, event);
        
        kafkaFuture.addCallback(
            future::complete,
            future::completeExceptionally
        );
        
        return future;
    }
}
```

```java
// infrastructure-adapter-output-events/src/main/java/com/pharmacare/infrastructure/adapter/output/events/config/KafkaProducerConfig.java
package com.pharmacare.infrastructure.adapter.output.events.config;

import com.pharmacare.domain.core.event.DomainEvent;
import com.pharmacare.infrastructure.adapter.output.events.serializer.DomainEventSerializer;
import lombok.RequiredArgsConstructor;
import org.apache.kafka.clients.producer.ProducerConfig;
import org.apache.kafka.common.serialization.StringSerializer;
import org.springframework.beans.factory.annotation.Value;
import org.springframework.context.annotation.Bean;
import org.springframework.context.annotation.Configuration;
import org.springframework.kafka.core.DefaultKafkaProducerFactory;
import org.springframework.kafka.core.KafkaTemplate;
import org.springframework.kafka.core.ProducerFactory;
import org.springframework.scheduling.concurrent.ThreadPoolTaskExecutor;

import java.util.HashMap;
import java.util.Map;
import java.util.concurrent.Executor;

/**
 * Configuração do produtor Kafka.
 */
@Configuration
@RequiredArgsConstructor
public class KafkaProducerConfig {
    
    @Value("${spring.kafka.bootstrap-servers}")
    private String bootstrapServers;
    
    @Value("${spring.kafka.producer.retries:3}")
    private int retries;
    
    @Value("${spring.kafka.producer.acks:all}")
    private String acks;
    
    @Value("${spring.kafka.producer.batch-size:16384}")
    private int batchSize;
    
    @Value("${spring.kafka.producer.linger-ms:1}")
    private int lingerMs;
    
    @Value("${spring.kafka.producer.buffer-memory:33554432}")
    private int bufferMemory;
    
    @Value("${spring.kafka.producer.compression-type:snappy}")
    private String compressionType;
    
    @Value("${spring.kafka.producer.enable-idempotence:true}")
    private boolean enableIdempotence;
    
    @Value("${spring.kafka.producer.max-in-flight-requests-per-connection:5}")
    private int maxInFlightRequests;
    
    @Bean
    public ProducerFactory<String, DomainEvent> producerFactory() {
        Map<String, Object> configProps = new HashMap<>();
        
        // Configurações básicas
        configProps.put(ProducerConfig.BOOTSTRAP_SERVERS_CONFIG, bootstrapServers);
        configProps.put(ProducerConfig.KEY_SERIALIZER_CLASS_CONFIG, StringSerializer.class);
        configProps.put(ProducerConfig.VALUE_SERIALIZER_CLASS_CONFIG, DomainEventSerializer.class);
        
        // Configurações de confiabilidade
        configProps.put(ProducerConfig.RETRIES_CONFIG, retries);
        configProps.put(ProducerConfig.ACKS_CONFIG, acks);
        configProps.put(ProducerConfig.ENABLE_IDEMPOTENCE_CONFIG, enableIdempotence);
        configProps.put(ProducerConfig.MAX_IN_FLIGHT_REQUESTS_PER_CONNECTION, maxInFlightRequests);
        
        // Configurações de desempenho
        configProps.put(ProducerConfig.BATCH_SIZE_CONFIG, batchSize);
        configProps.put(ProducerConfig.LINGER_MS_CONFIG, lingerMs);
        configProps.put(ProducerConfig.BUFFER_MEMORY_CONFIG, bufferMemory);
        configProps.put(ProducerConfig.COMPRESSION_TYPE_CONFIG, compressionType);
        
        // Configurações de timeout
        configProps.put(ProducerConfig.DELIVERY_TIMEOUT_MS_CONFIG, 120000); // 2 minutos
        configProps.put(ProducerConfig.REQUEST_TIMEOUT_MS_CONFIG, 30000); // 30 segundos
        
        // Configurações específicas para idempotência
        if (enableIdempotence) {
            configProps.put(ProducerConfig.MAX_IN_FLIGHT_REQUESTS_PER_CONNECTION, 5);
            configProps.put(ProducerConfig.RETRIES_CONFIG, Integer.MAX_VALUE);
        }
        
        // Configurações de métricas
        configProps.put(ProducerConfig.METRICS_RECORDING_LEVEL_CONFIG, "DEBUG");
        configProps.put(ProducerConfig.METRICS_SAMPLE_WINDOW_MS_CONFIG, 30000);
        
        return new DefaultKafkaProducerFactory<>(configProps);
    }
    
    @Bean
    public KafkaTemplate<String, DomainEvent> kafkaTemplate() {
        KafkaTemplate<String, DomainEvent> template = new KafkaTemplate<>(producerFactory());
        
        // Configurações adicionais do template
        template.setObservationEnabled(true); // Para Micrometer Tracing
        template.setProducerListener(new org.springframework.kafka.support.LoggingProducerListener<>());
        
        return template;
    }
    
    @Bean("kafkaEventPublisherExecutor")
    public Executor kafkaEventPublisherExecutor() {
        ThreadPoolTaskExecutor executor = new ThreadPoolTaskExecutor();
        executor.setCorePoolSize(5);
        executor.setMaxPoolSize(20);
        executor.setQueueCapacity(100);
        executor.setThreadNamePrefix("kafka-event-publisher-");
        executor.setRejectedExecutionHandler(
            new java.util.concurrent.ThreadPoolExecutor.CallerRunsPolicy()
        );
        executor.initialize();
        return executor;
    }
    
    /**
     * AdminClient para operações administrativas.
     */
    @Bean
    public org.apache.kafka.clients.admin.AdminClient adminClient() {
        Map<String, Object> configs = new HashMap<>();
        configs.put(ProducerConfig.BOOTSTRAP_SERVERS_CONFIG, bootstrapServers);
        return org.apache.kafka.clients.admin.AdminClient.create(configs);
    }
}
```

```java
// infrastructure-adapter-output-events/src/main/java/com/pharmacare/infrastructure/adapter/output/events/config/KafkaTopicConfig.java
package com.pharmacare.infrastructure.adapter.output.events.config;

import org.apache.kafka.clients.admin.AdminClientConfig;
import org.apache.kafka.clients.admin.NewTopic;
import org.springframework.beans.factory.annotation.Value;
import org.springframework.context.annotation.Bean;
import org.springframework.context.annotation.Configuration;
import org.springframework.kafka.core.KafkaAdmin;

import java.util.HashMap;
import java.util.Map;

/**
 * Configuração automática de tópicos Kafka.
 */
@Configuration
public class KafkaTopicConfig {
    
    @Value("${spring.kafka.bootstrap-servers}")
    private String bootstrapServers;
    
    @Value("${kafka.topics.product.events:product.events}")
    private String productEventsTopic;
    
    @Value("${kafka.topics.inventory.events:inventory.events}")
    private String inventoryEventsTopic;
    
    @Value("${kafka.topics.dlq.suffix:.dlq}")
    private String dlqSuffix;
    
    @Value("${kafka.topics.partitions:3}")
    private int partitions;
    
    @Value("${kafka.topics.replication-factor:1}")
    private short replicationFactor;
    
    @Value("${kafka.topics.retention-ms:604800000}") // 7 dias
    private long retentionMs;
    
    @Value("${kafka.topics.retention-bytes:1073741824}") // 1GB
    private long retentionBytes;
    
    @Bean
    public KafkaAdmin kafkaAdmin() {
        Map<String, Object> configs = new HashMap<>();
        configs.put(AdminClientConfig.BOOTSTRAP_SERVERS_CONFIG, bootstrapServers);
        configs.put(AdminClientConfig.REQUEST_TIMEOUT_MS_CONFIG, 5000);
        return new KafkaAdmin(configs);
    }
    
    @Bean
    public NewTopic productEventsTopic() {
        return createTopic(productEventsTopic);
    }
    
    @Bean
    public NewTopic inventoryEventsTopic() {
        return createTopic(inventoryEventsTopic);
    }
    
    @Bean
    public NewTopic productEventsDlqTopic() {
        return createDlqTopic(productEventsTopic);
    }
    
    @Bean
    public NewTopic inventoryEventsDlqTopic() {
        return createDlqTopic(inventoryEventsTopic);
    }
    
    private NewTopic createTopic(String topicName) {
        Map<String, String> configs = new HashMap<>();
        configs.put("retention.ms", String.valueOf(retentionMs));
        configs.put("retention.bytes", String.valueOf(retentionBytes));
        configs.put("cleanup.policy", "delete");
        configs.put("compression.type", "snappy");
        
        return new NewTopic(topicName, partitions, replicationFactor)
            .configs(configs);
    }
    
    private NewTopic createDlqTopic(String originalTopic) {
        Map<String, String> configs = new HashMap<>();
        configs.put("retention.ms", String.valueOf(retentionMs * 2)); // DLQ retém por mais tempo
        configs.put("retention.bytes", String.valueOf(retentionBytes * 2));
        configs.put("cleanup.policy", "delete");
        configs.put("compression.type", "none"); // DLQ não precisa de compressão
        
        String dlqTopicName = originalTopic + dlqSuffix;
        return new NewTopic(dlqTopicName, partitions, replicationFactor)
            .configs(configs);
    }
}
```

```java
// infrastructure-adapter-output-events/src/main/java/com/pharmacare/infrastructure/adapter/output/events/config/RetryConfig.java
package com.pharmacare.infrastructure.adapter.output.events.config;

import org.springframework.context.annotation.Configuration;
import org.springframework.retry.annotation.EnableRetry;
import org.springframework.scheduling.annotation.EnableAsync;

/**
 * Configuração de retentativas e processamento assíncrono.
 */
@Configuration
@EnableRetry
@EnableAsync
public class RetryConfig {
    
    /**
     * Retry template para operações Kafka.
     */
    @Bean
    public org.springframework.retry.support.RetryTemplate kafkaRetryTemplate() {
        return org.springframework.retry.support.RetryTemplate.builder()
            .maxAttempts(3)
            .exponentialBackoff(1000, 2, 10000) // 1s, 2s, 4s, 8s, max 10s
            .retryOn(org.springframework.kafka.KafkaException.class)
            .retryOn(java.io.IOException.class)
            .traversingCauses()
            .build();
    }
    
    /**
     * Listener para logging de retentativas.
     */
    @Bean
    public org.springframework.retry.RetryListener retryListener() {
        return new org.springframework.retry.RetryListener() {
            @Override
            public <T, E extends Throwable> void onError(
                org.springframework.retry.RetryContext context,
                org.springframework.retry.RetryCallback<T, E> callback,
                Throwable throwable
            ) {
```

```java
package com.pharmacare.infrastructure.adapter.output.events.metrics;

import io.micrometer.core.instrument.Counter;
import io.micrometer.core.instrument.MeterRegistry;
import io.micrometer.core.instrument.Timer;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.springframework.stereotype.Component;

import javax.annotation.PostConstruct;
import java.util.concurrent.ConcurrentHashMap;
import java.util.concurrent.TimeUnit;

/**
 * Componente para métricas de eventos Kafka.
 */
@Component
@RequiredArgsConstructor
@Slf4j
public class KafkaEventMetrics {
    
    private final MeterRegistry meterRegistry;
    
    private Counter totalPublishedEvents;
    private Counter totalFailedEvents;
    private Counter totalDlqEvents;
    private Timer publishLatency;
    
    private final ConcurrentHashMap<String, Counter> eventTypeCounters = new ConcurrentHashMap<>();
    private final ConcurrentHashMap<String, Counter> topicCounters = new ConcurrentHashMap<>();
    
    @PostConstruct
    public void init() {
        // Métricas globais
        totalPublishedEvents = Counter.builder("kafka.events.published.total")
            .description("Total de eventos publicados no Kafka")
            .register(meterRegistry);
        
        totalFailedEvents = Counter.builder("kafka.events.failed.total")
            .description("Total de eventos que falharam ao publicar")
            .register(meterRegistry);
        
        totalDlqEvents = Counter.builder("kafka.events.dlq.total")
            .description("Total de eventos enviados para DLQ")
            .register(meterRegistry);
        
        publishLatency = Timer.builder("kafka.events.publish.latency")
            .description("Latência de publicação de eventos")
            .publishPercentiles(0.5, 0.95, 0.99)
            .register(meterRegistry);
        
        // Métricas de produtor
        Counter.builder("kafka.producer.record.send.total")
            .description("Total de registros enviados pelo produtor")
            .register(meterRegistry);
        
        Counter.builder("kafka.producer.record.error.total")
            .description("Total de erros do produtor")
            .register(meterRegistry);
    }
    
    /**
     * Incrementa contador de sucesso.
     */
    public void incrementSuccess(String eventType, String topic) {
        totalPublishedEvents.increment();
        
        // Contador por tipo de evento
        String eventTypeKey = "kafka.events.published." + eventType.replace(".", "_");
        eventTypeCounters.computeIfAbsent(eventTypeKey, key ->
            Counter.builder(key)
                .description("Eventos publicados do tipo " + eventType)
                .tag("event_type", eventType)
                .register(meterRegistry)
        ).increment();
        
        // Contador por tópico
        String topicKey = "kafka.events.topic." + topic.replace(".", "_");
        topicCounters.computeIfAbsent(topicKey, key ->
            Counter.builder(key)
                .description("Eventos publicados no tópico " + topic)
                .tag("topic", topic)
                .register(meterRegistry)
        ).increment();
    }
    
    /**
     * Incrementa contador de falha.
     */
    public void incrementFailure(String eventType, String topic) {
        totalFailedEvents.increment();
        
        String failureKey = "kafka.events.failed." + eventType.replace(".", "_");
        Counter.builder(failureKey)
            .description("Eventos que falharam do tipo " + eventType)
            .tag("event_type", eventType)
            .tag("topic", topic)
            .register(meterRegistry)
            .increment();
    }
    
    /**
     * Incrementa contador de DLQ.
     */
    public void incrementDlq(String eventType, String dlqTopic) {
        totalDlqEvents.increment();
        
        String dlqKey = "kafka.events.dlq." + eventType.replace(".", "_");
        Counter.builder(dlqKey)
            .description("Eventos enviados para DLQ do tipo " + eventType)
            .tag("event_type", eventType)
            .tag("dlq_topic", dlqTopic)
            .register(meterRegistry)
            .increment();
    }
    
    /**
     * Registra latência de publicação.
     */
    public void recordPublishLatency(long duration, TimeUnit unit) {
        publishLatency.record(duration, unit);
    }
    
    /**
     * Registra tamanho do evento.
     */
    public void recordEventSize(String eventType, int sizeBytes) {
        meterRegistry.summary("kafka.events.size.bytes")
            .tag("event_type", eventType)
            .record(sizeBytes);
    }
    
    /**
     * Obtém métricas atuais.
     */
    public MetricsSnapshot getMetricsSnapshot() {
        return MetricsSnapshot.builder()
            .totalPublished(totalPublishedEvents.count())
            .totalFailed(totalFailedEvents.count())
            .totalDlq(totalDlqEvents.count())
            .meanLatencyMs(publishLatency.mean(TimeUnit.MILLISECONDS))
            .build();
    }
    
    @Data
    @Builder
    public static class MetricsSnapshot {
        private double totalPublished;
        private double totalFailed;
        private double totalDlq;
        private double meanLatencyMs;
    }
}
```

```java
import java.util.concurrent.TimeUnit;
import java.util.concurrent.TimeoutException;

/**
 * Health indicator para Kafka.
 * Verifica conectividade e status do cluster.
 */
@Component
@RequiredArgsConstructor
@Slf4j
public class KafkaHealthIndicator implements HealthIndicator {
    
    private final AdminClient adminClient;
    
    @Override
    public Health health() {
        try {
            // Verifica conectividade com o cluster
            DescribeClusterOptions options = new DescribeClusterOptions()
                .timeoutMs(5000);
            
            DescribeClusterResult clusterResult = adminClient.describeCluster(options);
            
            String clusterId = clusterResult.clusterId().get(3, TimeUnit.SECONDS);
            int nodeCount = clusterResult.nodes().get(3, TimeUnit.SECONDS).size();
            String controller = clusterResult.controller().get(3, TimeUnit.SECONDS).toString();
            
            log.debug("Health check Kafka - Cluster: {}, Nodes: {}, Controller: {}", 
                clusterId, nodeCount, controller);
            
            if (nodeCount == 0) {
                return Health.down()
                    .withDetail("message", "Nenhum nó Kafka disponível")
                    .withDetail("clusterId", clusterId)
                    .build();
            }
            
            // Health check adicional: verifica tópicos essenciais
            return Health.up()
                .withDetail("clusterId", clusterId)
                .withDetail("nodeCount", nodeCount)
                .withDetail("controller", controller)
                .withDetail("status", "CONNECTED")
                .build();
            
        } catch (InterruptedException e) {
            Thread.currentThread().interrupt();
            return Health.down()
                .withDetail("message", "Operação interrompida")
                .withException(e)
                .build();
                
        } catch (ExecutionException | TimeoutException e) {
            log.warn("Falha na conexão com Kafka: {}", e.getMessage());
            return Health.down()
                .withDetail("message", "Falha na conexão com Kafka")
                .withDetail("error", e.getMessage())
                .build();
                
        } catch (Exception e) {
            log.error("Erro inesperado no health check do Kafka", e);
            return Health.down()
                .withDetail("message", "Erro inesperado")
                .withException(e)
                .build();
        }
    }
}
```

## **17. Explicação da Aplicação**

### **17.1 Arquitetura Aplicada**

```
┌─────────────────────────────────────────────────────────┐
│               DOMÍNIO                                   │
│  ┌─────────────────────────────────────────────────┐   │
│  │           EventPublisherPort (Interface)        │   │

│                              │ (Dependência)            │
└─────────────────────────────────────────────────────────┘
                              │

┌─────────────────────────────────────────────────────────┐
│         INFRAESTRUTURA - ADAPTER OUTPUT                │
│  ┌─────────────────────────────────────────────────┐   │
│  │     EventPublisherKafkaAdapter                  │   │
│  │  (Implementa EventPublisherPort)                │   │
│  └─────────────────────────────────────────────────┘   │
│                              │                          │
│                    ┌─────────┴─────────┐                │
│                    ▼                   ▼                │

│                    ▼                   ▼                │
│           ┌──────────────┐   ┌──────────────────┐      │
│           │   Cluster    │   │   Configuração   │      │
│           │    Kafka     │   │     Jackson      │      │
│           └──────────────┘   └──────────────────┘      │
└─────────────────────────────────────────────────────────┘
```

### **17.2 Princípios Aplicados**

1. **Hexagonal Architecture**:
   - Porta (`EventPublisherPort`) definida pelo domínio
   - Adapter (`EventPublisherKafkaAdapter`) implementa a porta
   - Domínio não conhece detalhes de implementação Kafka

2. **Resiliência**:
   - Retry com backoff exponencial (3 tentativas)
   - Dead Letter Queue para falhas persistentes
   - Timeouts configuráveis
   - Idempotência habilitada

3. **Observabilidade**:
   - Métricas com Micrometer (sucesso, falha, DLQ, latência)
   - Health checks com Spring Actuator
   - Logging estruturado com correlationId
   - Tracing com headers de mensagem

4. **Segurança**:
   - Suporte a SSL/TLS
   - Configuração de truststore/keystore
   - Headers de segurança opcionais

5. **Performance**:
   - Publicação assíncrona com thread pool
   - Compressão Snappy
   - Batching e linger.ms otimizados
   - Pool de conexões configurável

### **17.3 Fluxo de Publicação**

```
1. Domínio chama eventPublisher.publish(event)
2. Adapter resolve tópico baseado no tipo de evento
3. Adiciona correlationId do MDC e metadados
4. Serializa evento para JSON
5. Envia para Kafka com headers
6. Em caso de sucesso: incrementa métricas, log
7. Em caso de falha: retentativa (3x com backoff)
8. Após falhas persistentes: envia para DLQ
9. Em falha crítica: log em storage alternativo
```

### **17.4 Decisões de Design**

1. **Serialização JSON**: Simplicidade, debuggability, compatibilidade
2. **Tópicos por domínio**: `product.events`, `inventory.events`
3. **Retry exponencial**: 1s, 2s, 4s com máximo de 10s
4. **DLQ por tópico**: `product.events.dlq`, `inventory.events.dlq`
5. **Headers padronizados**: event-id, correlation-id, aggregate-id
6. **Métricas detalhadas**: Por tipo de evento, por tópico, latência
