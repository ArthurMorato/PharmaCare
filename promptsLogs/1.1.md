```yaml
version: '3.8'

services:
  # MongoDB para dados de domínio
  mongodb:
    image: mongo:6.0
    container_name: pharmacare-mongodb
    restart: unless-stopped
    ports:
      - "27017:27017"
    environment:
      MONGO_INITDB_ROOT_USERNAME: admin
      MONGO_INITDB_ROOT_PASSWORD: admin123
      MONGO_INITDB_DATABASE: pharmacare
    volumes:
      - mongodb_data:/data/db
      - ./init-mongo.js:/docker-entrypoint-initdb.d/init-mongo.js:ro
    networks:
      - pharmacare-network
    healthcheck:
      test: ["CMD", "mongosh", "--eval", "db.adminCommand('ping')"]
      interval: 10s
      timeout: 5s
      retries: 5

  # PostgreSQL para dados transacionais
  postgresql:
    image: postgres:15-alpine
    container_name: pharmacare-postgres
    restart: unless-stopped
    ports:
      - "5432:5432"
    environment:
      POSTGRES_USER: pharmacare
      POSTGRES_PASSWORD: pharmacare123
      POSTGRES_DB: pharmacare_transactional
    volumes:
      - postgresql_data:/var/lib/postgresql/data
      - ./init-postgres.sql:/docker-entrypoint-initdb.d/init-postgres.sql:ro
    networks:
      - pharmacare-network
    healthcheck:
      test: ["CMD-SHELL", "pg_isready -U pharmacare"]
      interval: 10s
      timeout: 5s
      retries: 5

  # Redis para cache
  redis:
    image: redis:7-alpine
    container_name: pharmacare-redis
    restart: unless-stopped
    ports:
      - "6379:6379"
    command: redis-server --requirepass redis123
    volumes:
      - redis_data:/data
    networks:
      - pharmacare-network
    healthcheck:
      test: ["CMD", "redis-cli", "ping"]
      interval: 10s
      timeout: 5s
      retries: 5

  # Kafka para mensageria
  zookeeper:
    image: confluentinc/cp-zookeeper:latest
    container_name: pharmacare-zookeeper
    restart: unless-stopped
    environment:
      ZOOKEEPER_CLIENT_PORT: 2181
      ZOOKEEPER_TICK_TIME: 2000
    ports:
      - "2181:2181"
    networks:
      - pharmacare-network
    healthcheck:
      test: ["CMD", "bash", "-c", "echo stat | nc localhost 2181"]
      interval: 30s
      timeout: 10s
      retries: 3

  kafka:
    image: confluentinc/cp-kafka:latest
    container_name: pharmacare-kafka
    restart: unless-stopped
    depends_on:
      - zookeeper
    ports:
      - "9092:9092"
      - "9093:9093"
    environment:
      KAFKA_BROKER_ID: 1
      KAFKA_ZOOKEEPER_CONNECT: zookeeper:2181
      KAFKA_ADVERTISED_LISTENERS: PLAINTEXT://kafka:9092,PLAINTEXT_HOST://localhost:9093
      KAFKA_LISTENER_SECURITY_PROTOCOL_MAP: PLAINTEXT:PLAINTEXT,PLAINTEXT_HOST:PLAINTEXT
      KAFKA_INTER_BROKER_LISTENER_NAME: PLAINTEXT
      KAFKA_OFFSETS_TOPIC_REPLICATION_FACTOR: 1
    volumes:
      - kafka_data:/var/lib/kafka/data
    networks:
      - pharmacare-network
    healthcheck:
      test: ["CMD", "kafka-topics", "--list", "--bootstrap-server", "localhost:9092"]
      interval: 30s
      timeout: 10s
      retries: 3

  # Keycloak para autenticação
  keycloak:
    image: quay.io/keycloak/keycloak:21.1.1
    container_name: pharmacare-keycloak
    restart: unless-stopped
    command: start-dev
    environment:
      KC_DB: postgres
      KC_DB_URL: jdbc:postgresql://postgresql:5432/keycloak
      KC_DB_USERNAME: pharmacare
      KC_DB_PASSWORD: pharmacare123
      KEYCLOAK_ADMIN: admin
      KEYCLOAK_ADMIN_PASSWORD: admin123
      KC_FEATURES: scripts
    ports:
      - "8081:8080"
    depends_on:
      postgresql:
        condition: service_healthy
    volumes:
      - ./keycloak/realm-export.json:/opt/keycloak/data/import/realm-export.json:ro
    networks:
      - pharmacare-network
    healthcheck:
      test: ["CMD", "curl", "-f", "http://localhost:8080/realms/master"]
      interval: 30s
      timeout: 10s
      retries: 3

  # Elasticsearch para busca
  elasticsearch:
    image: elasticsearch:8.11.0
    container_name: pharmacare-elasticsearch
    restart: unless-stopped
    environment:
      - discovery.type=single-node
      - ES_JAVA_OPTS=-Xms512m -Xmx512m
      - xpack.security.enabled=false
    ulimits:
      memlock:
```

## **5. INSTRUÇÕES DE INTEGRAÇÃO**

1. **Clone o repositório** e coloque os arquivos na estrutura definida.
2. **Execute o Docker Compose** para subir os serviços:
   ```bash
   docker-compose up -d
   ```
3. **Verifique se todos os serviços estão saudáveis**:
   ```bash
   docker-compose ps
   ```
4. **Importe o projeto Maven** na sua IDE (IntelliJ IDEA, Eclipse, VS Code).
5. **Configure os perfis do Spring Boot**:
   - Crie `application.yml` no módulo `infrastructure-web/src/main/resources/`
   - Configure as conexões com os serviços Docker

## **6. CONSIDERAÇÕES DE BACKWARD COMPATIBILITY**

- A estrutura multi-módulo permite evolução independente de cada módulo.
- As versões das dependências são centralizadas no parent POM para facilitar upgrades.
- Os serviços Docker usam versões específicas (não latest) para garantir
reprodutibilidade.
- Scripts de inicialização são idempotentes (podem ser executados múltiplas vezes).
